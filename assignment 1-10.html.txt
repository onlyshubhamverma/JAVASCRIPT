<script>
	var n1 = 1,  n2 = 1, next_num, i;
	var num = parseInt (prompt (" Enter the limit for Fibonacci Series "));
	document.write( "Fibonacci Series: ");
	for ( i = 1; i <= num; i++)
		{  
			// print the n1
			document.write (" <br> " +  n1); 
			// sum of n1 and n2 into the next_num
    			next_num = n1 + n2; 

   			n1 = n2; 
    			n2 = next_num; 
		}

</script>
